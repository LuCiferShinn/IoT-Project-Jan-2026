<script>
    Chart.defaults.color = "#ffffff";
    const baseOpts = {
        maintainAspectRatio: false,
        plugins: { legend: { display: false } },
        scales: {
            y: { grid: { color: 'rgba(255,255,255,0.05)' } },
            x: { grid: { display: false }, ticks: { maxRotation: 45, font: { size: 9 } } }
        }
    };

    // ── Map ──────────────────────────────────────────────────────────────
    const map = L.map("map").setView([4.5, 101.0], 8);
    L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png').addTo(map);

    async function loadMapSpots() {
        // ── Flight path dots (cyan) ──
        const res    = await fetch('/api/map_data');
        const points = await res.json();
        points.forEach(p => {
            L.circleMarker([p.latitude, p.longitude], {
                radius: 2, color: '#00f2ff', fillOpacity: 0.4, stroke: false
            }).addTo(map);
        });
        if (points.length > 0) map.panTo([points[0].latitude, points[0].longitude]);

    }

    // ── KPI ──────────────────────────────────────────────────────────────
    async function loadSummary() {
        const res  = await fetch('/api/summary');
        const data = await res.json();
        document.getElementById('totalVal').innerText = data.total_flights.toLocaleString();
        document.getElementById('altVal').innerText   = data.average_altitude;
        document.getElementById('velVal').innerText   = data.average_velocity;
    }

    // ── Inferred Airport Table ───────────────────────────────────────────
    async function loadDepartureTable() {
        try {
            const res  = await fetch('/api/inferred_airports');
            const data = await res.json();

            const tbody = document.getElementById('airportTable');

            if (!data || data.length === 0) {
                tbody.innerHTML = '<tr><td colspan="4" style="color:#607d8b">No inferred data yet.</td></tr>';
                return;
            }

            tbody.innerHTML = data.map(d => {
                const typeColor  = d.type === 'origin' ? '#ff9800' : '#e040fb';
                const typeLabel  = d.type === 'origin' ? '🛫 Origin' : '🛬 Dest';
                const airportDisplay = d.airport_name && d.airport_name !== 'Unknown'
                    ? `<span style="color:#e0e0e0">${d.airport_name}</span>`
                    : `<span style="color:#455a64">Unknown</span>`;
                return `
                <tr>
                    <td style="font-family:monospace;color:#4fc3f7;white-space:nowrap">${d.flight_number}</td>
                    <td style="color:${typeColor};white-space:nowrap">${typeLabel}</td>
                    <td>${airportDisplay}</td>
                    <td style="font-family:monospace;font-size:0.58rem;color:#ff9800;white-space:nowrap">
                        ${d.est_lat}, ${d.est_lon}
                    </td>
                </tr>`;
            }).join('');

        } catch (e) {
            document.getElementById('depStatus').textContent = "Error loading data.";
            console.error(e);
        }
    }

    // ── Charts ───────────────────────────────────────────────────────────
    async function loadCharts() {
        // Hourly
        const hourData = await fetch('/api/flights_per_hour').then(r => r.json());
        new Chart(document.getElementById('hourChart'), {
            type: 'line',
            data: {
                labels: hourData.hours,
                datasets: [{
                    data: hourData.counts, borderColor: '#00f2ff',
                    fill: true, backgroundColor: 'rgba(0,242,255,0.1)', tension: 0.4
                }]
            },
            options: baseOpts
        });

        // Country donut + table (for the donut chart only — table now uses departure_airports)
        const countryData = await fetch('/api/top_countries').then(r => r.json());
        new Chart(document.getElementById('countryChart'), {
            type: 'doughnut',
            data: {
                labels: countryData.countries.slice(0, 5),
                datasets: [{
                    data: countryData.counts.slice(0, 5),
                    backgroundColor: ['#9575cd','#4fc3f7','#81c784','#ffb74d','#e57373'],
                    borderWidth: 0
                }]
            },
            options: {
                ...baseOpts,
                plugins: { legend: { display: true, position: 'bottom', labels: { font: { size: 9 } } } }
            }
        });

        // Daily
        const dayData = await fetch('/api/flights_per_day').then(r => r.json());
        new Chart(document.getElementById('dayChart'), {
            type: 'line',
            data: {
                labels: dayData.days,
                datasets: [{ data: dayData.counts, borderColor: '#81c784', tension: 0.3 }]
            },
            options: baseOpts
        });

        // Altitude distribution
        const altData = await fetch('/api/altitude_distribution').then(r => r.json());
        new Chart(document.getElementById('altChart'), {
            type: 'bar',
            data: {
                labels: altData.ranges,
                datasets: [{ data: altData.counts, backgroundColor: '#4db6ac' }]
            },
            options: baseOpts
        });

        // Velocity distribution
        const velData = await fetch('/api/velocity_distribution').then(r => r.json());
        new Chart(document.getElementById('velChart'), {
            type: 'bar',
            data: {
                labels: velData.ranges,
                datasets: [{ data: velData.counts, backgroundColor: '#f06292' }]
            },
            options: baseOpts
        });
    }

    // ── Init ─────────────────────────────────────────────────────────────
    window.onload = () => {
        loadSummary();
        loadMapSpots();
        loadCharts();
        loadDepartureTable();
        setTimeout(() => map.invalidateSize(), 500);
    };
</script>
